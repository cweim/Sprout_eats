Implement swipe-down gesture to close review sheet:

1. Add touch event listeners to sheet-content:
   - touchstart: record startY
   - touchmove: calculate deltaY, apply transform translateY(deltaY) if dragging down
   - touchend: if deltaY > 100px, close sheet; else snap back

2. Visual feedback:
   - Sheet follows finger during drag
   - Add transition for snap-back animation

3. Implementation pattern:
```javascript
function setupSheetGestures(sheetEl) {
    const content = sheetEl.querySelector('.sheet-content');
    let startY = 0;
    let currentY = 0;
    let isDragging = false;

    content.addEventListener('touchstart', (e) => {
        // Only drag from top area (near handle)
        if (e.target.closest('.sheet-body')) return;
        startY = e.touches[0].clientY;
        isDragging = true;
        content.style.transition = 'none';
    });

    content.addEventListener('touchmove', (e) => {
        if (!isDragging) return;
        currentY = e.touches[0].clientY - startY;
        if (currentY > 0) {
            content.style.transform = `translateY(${currentY}px)`;
        }
    });

    content.addEventListener('touchend', () => {
        if (!isDragging) return;
        isDragging = false;
        content.style.transition = 'transform 0.3s ease';
        if (currentY > 100) {
            closeReviewSheet();
        }
        content.style.transform = '';
        currentY = 0;
    });
}
```
