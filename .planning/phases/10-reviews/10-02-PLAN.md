```javascript
function createStarRating(container, initialRating = 0, onChange = null) {
    container.innerHTML = '';
    for (let i = 1; i <= 5; i++) {
        const star = document.createElement('span');
        star.className = 'star';
        star.textContent = i <= initialRating ? '⭐' : '☆';
        star.dataset.value = i;
        star.addEventListener('click', () => {
            const newRating = i;
            container.dataset.rating = newRating;
            updateStars(container, newRating);
            if (onChange) onChange(newRating);
        });
        container.appendChild(star);
    }
}

function updateStars(container, rating) {
    container.querySelectorAll('.star').forEach((star, index) => {
        star.textContent = index < rating ? '⭐' : '☆';
    });
}
```

```javascript
const PRICE_LABELS = ['', 'Budget-friendly', 'Affordable', 'Moderate', 'Pricey', 'Splurge'];

function createPriceRating(container, labelEl, initialRating = 0, onChange = null) {
    container.innerHTML = '';
    for (let i = 1; i <= 5; i++) {
        const money = document.createElement('span');
        money.className = 'money';
        money.textContent = '💰';
        money.dataset.value = i;
        money.style.opacity = i <= initialRating ? '1' : '0.3';
        money.addEventListener('click', () => {
            const newRating = i;
            container.dataset.rating = newRating;
            updatePriceRating(container, labelEl, newRating);
            if (onChange) onChange(newRating);
        });
        container.appendChild(money);
    }
    labelEl.textContent = PRICE_LABELS[initialRating] || '';
}

function updatePriceRating(container, labelEl, rating) {
    container.querySelectorAll('.money').forEach((m, index) => {
        m.style.opacity = index < rating ? '1' : '0.3';
    });
    labelEl.textContent = PRICE_LABELS[rating] || '';
}
```

4. Add timestamp formatting function:
```javascript
function formatTimestamp(dateStr) {
    if (!dateStr) return '';
    const date = new Date(dateStr);
    const now = new Date();
    const diffMs = now - date;
    const diffMins = Math.floor(diffMs / 60000);
    const diffHours = Math.floor(diffMs / 3600000);
    const diffDays = Math.floor(diffMs / 86400000);

    if (diffMins < 1) return 'just now';
    if (diffMins < 60) return `${diffMins} minutes ago`;
    if (diffHours < 24) return `${diffHours} hours ago`;
    if (diffDays === 1) return 'yesterday';
    if (diffDays < 7) return `${diffDays} days ago`;
    return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
}
```

```javascript
let currentReviewPlaceId = null;
let currentReviewData = null;

async function openReviewSheet(placeId, placeName) {
    currentReviewPlaceId = placeId;
    document.getElementById('review-sheet-title').textContent = `Review: ${placeName}`;

    // Fetch existing review if any
    try {
        const response = await fetch(`${API_BASE}/api/places/${placeId}/review`);
        if (response.ok) {
            currentReviewData = await response.json();
            populateReviewSheet(currentReviewData);
            document.getElementById('delete-review-btn').style.display = 'block';
        } else {
            currentReviewData = null;
            resetReviewSheet();
            document.getElementById('delete-review-btn').style.display = 'none';
        }
    } catch (e) {
        currentReviewData = null;
        resetReviewSheet();
    }

    document.getElementById('review-sheet').style.display = 'flex';
    // Animate slide up
}

function closeReviewSheet() {
    document.getElementById('review-sheet').style.display = 'none';
    currentReviewPlaceId = null;
    currentReviewData = null;
}

function populateReviewSheet(review) {
    // Set overall ratings
    const overallStars = document.getElementById('overall-stars');
    createStarRating(overallStars, review.overall_rating);

    const priceRating = document.getElementById('price-rating');
    const priceLabel = document.getElementById('price-label');
    createPriceRating(priceRating, priceLabel, review.price_rating);

    document.getElementById('overall-remarks').value = review.overall_remarks || '';
    document.getElementById('overall-edited').textContent =
        review.updated_at ? 'edited ' + formatTimestamp(review.updated_at) : '';

    // Populate dishes
    const dishesContainer = document.getElementById('review-dishes');
    dishesContainer.innerHTML = '';
    if (review.dishes && review.dishes.length > 0) {
        review.dishes.forEach((dish, i) => {
            dishesContainer.insertAdjacentHTML('beforeend', createDishCard(dish, i));
            // Initialize star rating for this dish
            const card = dishesContainer.lastElementChild;
            createStarRating(card.querySelector('.dish-rating'), dish.rating);
        });
    }
}

function resetReviewSheet() {
    document.getElementById('review-dishes').innerHTML = '';
    createStarRating(document.getElementById('overall-stars'), 0);
    createPriceRating(document.getElementById('price-rating'),
                      document.getElementById('price-label'), 0);
    document.getElementById('overall-remarks').value = '';
    document.getElementById('overall-edited').textContent = '';
}
```

```javascript
async function saveReview() {
    const overallRating = parseInt(document.getElementById('overall-stars').dataset.rating);
    const priceRating = parseInt(document.getElementById('price-rating').dataset.rating);

    if (overallRating < 1) {
        showToast('Please add an overall rating');
        return;
    }
    if (priceRating < 1) {
        showToast('Please add a price rating');
        return;
    }

    // Collect dishes
    const dishes = [];
    document.querySelectorAll('.dish-card').forEach(card => {
        const name = card.querySelector('.dish-name-input').value.trim();
        const rating = parseInt(card.querySelector('.dish-rating').dataset.rating);
        const remarks = card.querySelector('.dish-remarks').value.trim();
        const id = card.dataset.dishId.startsWith('new-') ? null : parseInt(card.dataset.dishId);

        if (name && rating > 0) {
            dishes.push({ id, name, rating, remarks: remarks || null });
        }
    });

    const data = {
        overall_rating: overallRating,
        price_rating: priceRating,
        overall_remarks: document.getElementById('overall-remarks').value.trim() || null,
        dishes: dishes
    };

    try {
        const response = await fetch(`${API_BASE}/api/places/${currentReviewPlaceId}/review`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });

        if (response.ok) {
            showToast('Review saved!');
            closeReviewSheet();
            // Refresh place card to show review
            await loadPlaces();
        } else {
            showToast('Failed to save review');
        }
    } catch (e) {
        showToast('Error saving review');
    }
}
```
